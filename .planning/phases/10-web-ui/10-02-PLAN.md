      function shellRoot() {
        return {
          init() {
            // D-22: single tab-visibility listener on the whole shell.
            // On un-hide, fire an immediate health-strip refresh and let
            // htmx's own `every 10s[!document.hidden]` pause/resume cadence.
            document.addEventListener('visibilitychange', () => {
              if (!document.hidden) {
                const strip = document.getElementById('health-strip');
                if (strip) { htmx.trigger(strip, 'load'); }
              }
            });
            // D-11: `/` focuses the query-console input from anywhere
            // on the page, as long as the user isn't already typing in
            // another <input> / <textarea> / [contenteditable].
            document.addEventListener('keydown', (e) => {
              const tag = document.activeElement ? document.activeElement.tagName : '';
              const inEditable = tag === 'INPUT' || tag === 'TEXTAREA'
                || (document.activeElement && document.activeElement.isContentEditable);
              if (e.key === '/' && !inEditable) {
                const q = document.getElementById('query-console-input');
                if (q) { e.preventDefault(); q.focus(); }
              }
            });
          }
        }
      }
    </script>
    {% endblock %}

      <script>
        function queryConsole(opts) {
          return {
            tokenString: '',
            errorMessage: '',
            init() {
              // D-26: pre-populate input from URL params on first paint.
              this.tokenString = this.$refs.input.value;
            },
            parse(input) {
              // D-08 grammar: whitespace-separated key:value pairs.
              // Returns [params, errors]. errors is [] or [{token, reason}].
              const params = {};
              const errors = [];
              const tokens = (input || '').trim().split(/\s+/).filter(Boolean);
              for (const tok of tokens) {
                const idx = tok.indexOf(':');
                if (idx <= 0) {
                  errors.push({ token: tok, reason: 'missing key:value separator' });
                  continue;
                }
                const key = tok.substring(0, idx).toLowerCase();
                const val = tok.substring(idx + 1);
                if (opts.supported.length && !opts.supported.includes(key)) {
                  errors.push({ token: tok, reason: `unknown key "${key}"` });
                  continue;
                }
                // Map UI grammar → Phase 9 API query params (D-07: client does the translation)
                // tools view: origin, namespace, readonly, q
                // execs view: tool, since, until, promoted, hash
                // manifest view: q, status
                if (key === 'promoted') { params.promoted_only = val; }
                else if (key === 'hash') { params.code_hash = val; }
                else { params[key] = val; }
              }
              return [params, errors];
            },
            buildUrl(params) {
              const qs = new URLSearchParams(params).toString();
              const base = `/ui/${opts.view}`;
              return qs ? `${base}?${qs}` : base;
            },
            submit() {
              const [params, errors] = this.parse(this.tokenString);
              if (errors.length) {
                // D-10: inline error below input, never toast.
                const keys = opts.supported.join(', ');
                this.errorMessage = `Unknown filter key: \`${errors[0].token}\`. Supported keys: ${keys}.`;
                return;
              }
              this.errorMessage = '';
              const url = this.buildUrl(params);
              htmx.ajax('GET', url, {
                target: '#view-main',
                swap: 'innerHTML',
                pushUrl: true,
              });
            },
            clearAndReset() {
              this.tokenString = '';
              this.errorMessage = '';
              htmx.ajax('GET', `/ui/${opts.view}`, {
                target: '#view-main',
                swap: 'innerHTML',
                pushUrl: true,
              });
            },
            stuffAndSubmit(tokens) {
              this.tokenString = tokens;
              this.submit();
            },
          }
        }
      </script>
    </div>
