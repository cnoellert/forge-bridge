```python
"""Wire forge-bridge's learning pipeline into projekt-forge startup.

LRN-01: ExecutionLog gets a per-project log path (not the standalone default).
LRN-02: A storage callback mirrors executions to projekt-forge's logs. EXT-03
        (SQL backend) is deferred — the callback is currently a logger-only
        stub. When EXT-03 lands, replace _persist_execution's body with a real
        session.add + commit.
LRN-03: LLMRouter is constructed from forge_config.yaml's llm section and
        injected into SkillSynthesizer.
LRN-04: A pre_synthesis_hook returns a PreSynthesisContext enriching the
        synthesizer's prompts with projekt-forge constraints and tags.
"""
from __future__ import annotations

import logging
import os
from pathlib import Path
from typing import Any, Optional

from forge_bridge import (
    ExecutionLog,
    ExecutionRecord,
    LLMRouter,
    PreSynthesisContext,
    SkillSynthesizer,
)

logger = logging.getLogger("projekt_forge.learning.wiring")

# Module-level state — populated by init_learning_pipeline() at startup.
# Accessed by tests and by downstream call sites (e.g., MCP tools that need
# to reach the synthesizer directly).
_state: dict[str, Any] = {}


# ---------------------------------------------------------------------------
# Builders
# ---------------------------------------------------------------------------


def _build_llm_router(cfg: dict) -> LLMRouter:
    """Construct LLMRouter from forge_config.yaml's llm section.

    All keys are optional — missing keys fall through to LLMRouter's own
    env/default chain (D-13, D-14).
    """
    llm_cfg = cfg.get("llm") or {}
    router = LLMRouter(
        local_url=llm_cfg.get("local_url"),
        local_model=llm_cfg.get("local_model"),
        system_prompt=llm_cfg.get("system_prompt"),
    )
    logger.info(
        "LLMRouter configured: local_url=%s, local_model=%s",
        router.local_url,
        router.local_model,
    )
    return router


def _resolve_log_path(cfg: dict) -> Path:
    """Resolve ExecutionLog path by priority: cfg > FORGE_PROJECT_ROOT > default.

    Per D-15/D-16: per-process log isolation is a naming discipline. Standalone
    forge-bridge defaults to ~/.forge-bridge/executions.jsonl; projekt-forge
    writes to $FORGE_PROJECT_ROOT/.forge/executions.jsonl so two processes
    never share a log file.
    """
    learning_cfg = cfg.get("learning") or {}
    explicit = learning_cfg.get("log_path")
    if explicit:
        return Path(explicit)

    project_root = os.environ.get("FORGE_PROJECT_ROOT")
    if project_root:
        return Path(project_root) / ".forge" / "executions.jsonl"

    # Fallback — same as forge-bridge's own default. Deliberate: standalone
    # forge-bridge and projekt-forge will share the log only when neither env
    # var nor config key is set, which is a "local dev, one process at a time"
    # scenario and D-16 explicitly accepts this outcome.
    return Path.home() / ".forge-bridge" / "executions.jsonl"


def _build_execution_log(cfg: dict) -> ExecutionLog:
    """Construct ExecutionLog with a per-project log path and configured threshold."""
    learning_cfg = cfg.get("learning") or {}
    log_path = _resolve_log_path(cfg)
    threshold = int(learning_cfg.get("threshold", 3))
    log = ExecutionLog(log_path=log_path, threshold=threshold)
    logger.info("ExecutionLog configured: log_path=%s, threshold=%d", log_path, threshold)
    return log


# ---------------------------------------------------------------------------
# Hooks registered with forge-bridge
# ---------------------------------------------------------------------------


async def _persist_execution(record: ExecutionRecord) -> None:
    """Storage callback: mirror every ExecutionRecord to projekt-forge's logs.

    EXT-03 (SQL persistence backend) is deferred to v1.1.x. Today this is a
    logger-only stub that proves the callback fires end-to-end. When EXT-03
    lands, replace the body of this function with:

        from projekt_forge.db.engine import get_engine, get_session_factory
        engine = get_engine()
        Session = get_session_factory(engine)
        async with Session() as session:
            session.add(ExecutionRow.from_record(record))
            await session.commit()

    The contract is stable (ExecutionRecord fields are locked by D-03); only
    the storage mechanism is swap-able.
    """
    logger.info(
        "execution mirrored: code_hash=%s intent=%r promoted=%s",
        record.code_hash[:12],
        record.intent,
        record.promoted,
    )


async def _build_pre_synthesis_context(intent: str, params: dict) -> PreSynthesisContext:
    """pre_synthesis_hook: enrich synthesis prompts with projekt-forge context.

    Today this returns a minimal context (the hard "do not import flame"
    constraint plus an optional project tag). When projekt-forge grows a
    richer project-context layer, read active shot / roles / naming
    conventions here and plumb them into extra_context.
    """
    constraints = ["do not import flame"]
    tags: list[str] = []

    project_code = os.environ.get("FORGE_PROJECT_CODE")
    if project_code:
        tags.append(f"project:{project_code}")

    return PreSynthesisContext(
        extra_context="",
        tags=tags,
        examples=[],
        constraints=constraints,
    )


# ---------------------------------------------------------------------------
# Public orchestrator
# ---------------------------------------------------------------------------


def init_learning_pipeline(args: Any = None) -> dict[str, Any]:
    """Construct LLMRouter + ExecutionLog + SkillSynthesizer and register hooks.

    Idempotent: subsequent calls replace the stored state (callers get the
    latest construction). This matters for tests that re-run startup under
    fresh config.

    Args:
        args: argparse.Namespace from projekt_forge.__main__ (currently unused
              but accepted for forward compatibility — future CLI flags may
              override config values).

    Returns:
        dict with keys: router, execution_log, synthesizer. Also stored in
        module-level _state for downstream retrieval via get_* accessors.
    """
    # Load config; tolerate missing config file (falls through to env/defaults).
    from projekt_forge.config.forge_config import load_forge_config

    try:
        cfg = load_forge_config()
    except RuntimeError as exc:
        logger.warning("forge_config unavailable (%s) — using env/default fallbacks", exc)
        cfg = {}

    # Build components.
    router = _build_llm_router(cfg)
    execution_log = _build_execution_log(cfg)

    # Wire hooks.
    execution_log.set_storage_callback(_persist_execution)
    synthesizer = SkillSynthesizer(
        router=router,
        pre_synthesis_hook=_build_pre_synthesis_context,
    )

    _state["router"] = router
    _state["execution_log"] = execution_log
    _state["synthesizer"] = synthesizer

    logger.info(
        "learning pipeline initialized (model=%s, log_path=%s)",
        router.local_model,
        execution_log._path,
    )
    return dict(_state)


# ---------------------------------------------------------------------------
# Accessors (for tests and downstream call sites)
# ---------------------------------------------------------------------------


def get_execution_log() -> Optional[ExecutionLog]:
    """Return the ExecutionLog constructed by init_learning_pipeline(), or None."""
    return _state.get("execution_log")


def get_synthesizer() -> Optional[SkillSynthesizer]:
    """Return the SkillSynthesizer constructed by init_learning_pipeline(), or None."""
    return _state.get("synthesizer")


def get_router() -> Optional[LLMRouter]:
    """Return the LLMRouter constructed by init_learning_pipeline(), or None."""
    return _state.get("router")


def _reset_for_testing() -> None:
    """Clear the module-level state. Test-only."""
    _state.clear()
```
