```python
async def _lifespan(mcp_server: FastMCP):
    """Server lifespan: connect client, start watcher, clean up on exit."""
    global _server_started
    await startup_bridge()
    _server_started = True  # D-14: trips the register_tools() guard

        await shutdown_bridge()
        _server_started = False  # reset for clean teardown / test isolation


async def startup_bridge(
    server_url: str | None = None,
    client_name: str | None = None,
) -> None:
    """Connect to forge-bridge server before serving MCP requests.

    Args:
        server_url: WebSocket URL of the forge-bridge server.
                    Defaults to $FORGE_BRIDGE_URL or ws://127.0.0.1:9998.
        client_name: Identifier for this MCP client.
                     Defaults to $FORGE_MCP_CLIENT_NAME or mcp_claude.
    """
    global _client
    server_url = server_url or os.environ.get("FORGE_BRIDGE_URL", "ws://127.0.0.1:9998")
    client_name = client_name or os.environ.get("FORGE_MCP_CLIENT_NAME", "mcp_claude")

    _client = AsyncClient(
        client_name=client_name,
        server_url=server_url,
        endpoint_type="mcp",
        auto_reconnect=True,
    )
    await _client.start()
    try:
        await _client.wait_until_connected(timeout=10.0)
        logger.info(f"Connected to forge-bridge at {server_url}")
    except Exception as e:
        logger.warning(
            f"Could not connect to forge-bridge at {server_url}: {e}\n"
            "forge_* tools will fail. flame_* tools still work if Flame is running."
        )


async def shutdown_bridge() -> None:
    """Disconnect from forge-bridge server and clean up."""
    global _client
    if _client:
        await _client.stop()
        _client = None
```

```python
# AFTER
def register_tools(
    mcp: FastMCP,
    fns: list[Callable[..., Any]],
    prefix: str = "",
    source: str = "user-taught",
) -> None:
    """Register multiple tools under a shared prefix and source tag.

    Public API for downstream consumers (e.g. projekt-forge).

    Usage (before mcp.run()):
        from forge_bridge.mcp import register_tools, get_mcp
        register_tools(get_mcp(), [my_fn1, my_fn2], prefix="forge_")

    Raises:
        RuntimeError: If called after mcp.run() has started. All tool
                      registration must happen before the server starts.

    Args:
        mcp:    The live FastMCP instance.
        fns:    List of callables to register.
        prefix: Prefix prepended to each fn.__name__ to form the tool name.
        source: Source tag for all tools in this batch.
    """
    # Lazy import avoids the server.py -> registry.py -> server.py cycle.
    # Accessing _server_started through the module captures the *current*
    # value, not a stale snapshot (see RESEARCH.md R-5).
    import forge_bridge.mcp.server as _server
    if _server._server_started:
        raise RuntimeError(
            "register_tools() cannot be called after the MCP server has started. "
            "Register all tools before calling mcp.run()."
        )

    for fn in fns:
        name = f"{prefix}{fn.__name__}" if prefix else fn.__name__
        register_tool(mcp, fn, name=name, source=source)
```

```python
def test_register_tools_builtin_source():
    """PKG-01: register_tools(source='builtin') is accepted and the tool's
    _source metadata is 'builtin' for downstream builtin callers."""
    mcp = _fresh_mcp()

    def my_tool() -> str:
        """A downstream-supplied tool."""
        return "ok"

    register_tools(mcp, [my_tool], prefix="forge_", source="builtin")

    # Tool registered under the expected name and carries source=builtin metadata
    registered = mcp._tool_manager._tools
    assert "forge_my_tool" in registered
    tool = registered["forge_my_tool"]
    # FastMCP stores meta on the tool object; check meta._source
    assert tool.meta == {"_source": "builtin"}


def test_register_tools_post_run_guard():
    """API-05: register_tools raises RuntimeError when _server_started is True."""
    import forge_bridge.mcp.server as server_mod
    mcp = _fresh_mcp()

    def my_tool() -> str:
        """Late tool."""
        return "late"

    original = server_mod._server_started
    try:
        server_mod._server_started = True
        with pytest.raises(RuntimeError, match="cannot be called after the MCP server has started"):
            register_tools(mcp, [my_tool], prefix="forge_")
    finally:
        server_mod._server_started = original


def test_register_tools_pre_run_ok():
    """API-05: register_tools succeeds when _server_started is False (default)."""
    import forge_bridge.mcp.server as server_mod
    mcp = _fresh_mcp()

    def my_tool() -> str:
        """Early tool."""
        return "early"

    original = server_mod._server_started
    try:
        server_mod._server_started = False
        register_tools(mcp, [my_tool], prefix="forge_")  # must not raise
        assert "forge_my_tool" in mcp._tool_manager._tools
    finally:
        server_mod._server_started = original
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-06 | Elevation of Privilege | `register_tools()` post-`run()` bypass via race condition between registration and server start | mitigate | The `_server_started` flag is set synchronously inside `_lifespan` BEFORE `yield` (the point at which FastMCP starts accepting MCP client connections). Registration attempts that happen during the same event-loop iteration as the lifespan transition would see `False`, but by the time any MCP client is connected and could trigger tool discovery, the flag is True. This is not a true race because `_lifespan` is the single writer and runs inside FastMCP's own lifecycle coordinator. For ASVS L1, this is sufficient; a threading.Lock would be overengineering for a single-event-loop server. |
| T-04-07 | Tampering | Downstream caller sets `_server_started = False` manually to bypass the guard | accept | `_server_started` is a single-underscore name (convention = private). Python does not enforce privacy; a determined consumer can always mutate it. Defense-in-depth via double-underscore name mangling or read-only descriptor adds complexity for no real security gain — the guard exists to catch programming errors, not to defend against adversarial consumers. CONVENTIONS.md's `_` prefix convention communicates intent. |
| T-04-08 | Information Disclosure | `startup_bridge(server_url=<logged-in-error>)` writes the injected URL to log output | accept | `logger.info(f"Connected to forge-bridge at {server_url}")` and the warning path both emit the URL. Existing behavior; not a regression introduced by this plan. If a consumer passes a URL containing credentials (`ws://user:pass@host`), those credentials land in logs — but that's a misuse pattern, not a library bug. Document in Phase 6 integration guide that URLs passed to `startup_bridge` should not embed credentials. |
| T-04-09 | Denial of Service | `startup_bridge(server_url="ws://unreachable:9998")` hangs for 10s on `wait_until_connected(timeout=10.0)` | accept | 10-second timeout is hardcoded at line 127 (pre-existing). The try/except around `wait_until_connected` logs a warning and continues — the server starts with `_client` attached but disconnected, and `auto_reconnect=True` handles recovery. No change to existing behavior. |
| T-04-10 | Tampering | Circular import between `server.py` ↔ `registry.py` could cause one module to read a half-initialized `_server_started` | mitigate | Lazy `import forge_bridge.mcp.server as _server` inside `register_tools` defers the module-load until first call. At that point `server.py` is fully loaded (since the MCP lifespan must have run to set `_server_started = True` in the first place). For cold calls before server start, `_server_started` reads `False` correctly because module-scope `_server_started: bool = False` initializes before any `register_tools` call from the `register_builtins` path. See RESEARCH.md R-5 for full analysis. |
