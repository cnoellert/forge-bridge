    from pathlib import Path

    _SYNTH_ROOT = Path(os.environ.get(

    def _tools_preset_chips() -> list:
        return [
            {"label": "Active synth", "tokens": "origin:synthesized"},
            {"label": "Quarantined", "tokens": "origin:synthesized q:quarantined"},
            {"label": "Builtin only", "tokens": "origin:builtin"},
        ]


    def _filter_tools(tools, qp):
        """Apply UI-grammar filters to get_tools() result. v1.3 does filtering
        in-Python because /api/v1/tools has no server-side filter params (D-23)."""
        origin = qp.get("origin")
        namespace = qp.get("namespace")
        readonly = qp.get("readonly")
        q = (qp.get("q") or "").lower()
        out = []
        for t in tools:
            if origin and t.origin != origin:
                continue
            if namespace and t.namespace != namespace:
                continue
            if readonly is not None:
                m = dict(t.meta) if t.meta else {}
                if str(m.get("read_only_hint", "")).lower() != readonly.lower():
                    continue
            if q and q not in t.name.lower():
                continue
            out.append(t)
        return out


    def _read_synth_source(name: str) -> str | None:
        """Best-effort: read raw source for a synthesized tool. Returns None
        if the file is missing or unreadable."""
        candidate = _SYNTH_ROOT / f"{name}.py"
        try:
            if candidate.is_file():
                return candidate.read_text(encoding="utf-8")
        except (OSError, UnicodeDecodeError) as exc:
            logger.warning(
                "_read_synth_source failed for %s: %s",
                name, type(exc).__name__, exc_info=True,
            )
        return None


    async def ui_tools_handler(request: Request) -> HTMLResponse:

            return _render_error(

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL query params ↔ filter logic | `origin`, `namespace`, `readonly`, `q` params filter get_tools() result; malformed values must not crash |
| Disk filesystem ↔ raw source reader | `_read_synth_source` reads `~/.forge-bridge/tools/synth/<name>.py`; path must be confined |
| Tool name ↔ filesystem | `/ui/tools/{name}` drilldown — `name` is used as filename suffix for raw source read |
| ToolRecord fields ↔ Jinja templates | Tool names, consumer tags, meta values render into HTML — autoescape is the defense |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-18 | Tampering | Path traversal via `/ui/tools/../../../etc/passwd` | mitigate | Starlette's path-parameter extraction normalizes the `{name}` placeholder and rejects segments containing `/`. `_read_synth_source` constructs `_SYNTH_ROOT / f"{name}.py"` — `Path` normalizes but doesn't fully prevent traversal. Add explicit guard: verify the resolved path still has `_SYNTH_ROOT` as a parent before reading. |
| T-10-19 | Information Disclosure | XSS via malicious tool name / consumer tag | mitigate | Autoescape on (Jinja2Templates default); tool names and tags flow through `{{ }}` interpolation which HTML-escapes. Consumer tags additionally pass Phase 7 `_sanitize_tag` at read time (upstream belt). |
