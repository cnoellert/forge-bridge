    (port, fake_api) so the test can hit http://127.0.0.1:{port}/ui/tools

    finally:

        if task is not None:

                await asyncio.wait_for(task, timeout=5.0)
            except (asyncio.TimeoutError, asyncio.CancelledError, Exception):
                task.cancel()
                try:
                    await task
                except Exception:
                    pass


@pytest.mark.asyncio
async def test_nav_click_yields_single_nav_and_single_health_strip(
    live_console_server,
) -> None:
    """D-39 invariant: after a boosted nav click, the rendered DOM must
    contain exactly one `.top-nav` element and exactly one `#health-strip`
    element. This test fails loudly against the PRE-fix `shell.html` state
    (where hx-target="#view-main" + hx-swap="innerHTML" cause a duplicate).
    """
    pytest.importorskip("playwright.async_api")
    from playwright.async_api import async_playwright

    port, _api = live_console_server

    async with async_playwright() as p:
        browser = await p.chromium.launch()
        try:
            page = await browser.new_page()
            await page.goto(f"http://127.0.0.1:{port}/ui/tools")

            # Sanity: initial page has exactly one of each
            initial_nav_count = await page.locator("nav.top-nav").count()
            initial_strip_count = await page.locator("#health-strip").count()
            assert initial_nav_count == 1, (
                f"Initial page load has {initial_nav_count} `.top-nav` elements; expected 1. "
                "Server-rendered shell.html is malformed before any htmx interaction."
            )
            assert initial_strip_count == 1, (
                f"Initial page load has {initial_strip_count} `#health-strip` elements; expected 1."
            )

            # Click a nav link (Executions) — triggers hx-boost swap
            await page.locator('a.nav-link[href="/ui/execs"]').click()

            # htmx-boost swap settle: wait for the boosted request to complete + DOM to update.
            # networkidle is the safe default for boosted navigation since hx-push-url rewrites
            # the URL and htmx swaps the body synchronously once the response lands.
            await page.wait_for_load_state("networkidle")

            # D-39 invariant assertions
            post_nav_count = await page.locator("nav.top-nav").count()
            post_strip_count = await page.locator("#health-strip").count()
            assert post_nav_count == 1, (
                f"After nav click, rendered DOM has {post_nav_count} `.top-nav` elements; "
                "expected exactly 1. This is the 2026-04-23 duplicate-nav regression — the "
                "`hx-boost` + `hx-target=#view-main` + `hx-swap=innerHTML` mismatch is back "
                "on shell.html. Fix: remove `hx-target` and `hx-swap` from the `<nav>` element."
            )
            assert post_strip_count == 1, (
                f"After nav click, rendered DOM has {post_strip_count} `#health-strip` "
                "elements; expected exactly 1. Duplicate-health-strip is the same bug class "
                "as the duplicate-nav — check shell.html hx-boost swap configuration."
            )

            # Belt-and-suspenders: URL was actually pushed (hx-push-url contract)
            assert page.url.endswith("/ui/execs"), (
                f"Expected URL /ui/execs after nav click; got {page.url}. "
                "hx-push-url contract may be broken."
            )
        finally:
            await browser.close()

## Deviations from Plan

### Auto-fixed Issues

**1. [Rule 3 — Blocking] D-43 verification cycle used `git checkout` round-trip instead of `git stash push/pop`**

- **Found during:** Task 2, Step 1 of the pre-commit verification cycle.
- **Issue:** The plan's exact action text specifies `git stash push -m "10.1-04 D-43 pre-fix restore" forge_bridge/console/templates/shell.html` as Step 1 of the cycle. `git stash push` captures the working-tree diff for a file against the index. In this worktree shell.html is in the committed post-fix state with no working-tree modification — so `git stash push forge_bridge/console/templates/shell.html` reports `No local changes to save` and fails to create a stash entry. The plan's own Step 1 note ("If `git stash push` reports 'No local changes to save', STOP — it means shell.html was not in the post-fix committed state") does not apply here: shell.html IS in the post-fix state; the failure mode is that the plan's mental model assumes the Plan 10.1-02 fix is still in the working tree when it is in fact committed upstream.
- **Why it blocks:** Without a mechanism to put the pre-fix buggy content on disk, the D-43 verification cycle cannot be performed, and the SUMMARY cannot record the FAIL evidence — which is the plan's load-bearing content.
- **Fix:** Used a `git checkout <pre-fix-commit> -- shell.html` → run test (FAIL) → `git checkout HEAD -- shell.html` → run test (PASS) round-trip, which is functionally equivalent: both reveal the pre-fix state in the working tree, let the test execute against it, then restore HEAD. The pre-fix commit for shell.html is `78daa3f` (`feat(10-02): add shell.html app shell with nav, health strip, Alpine root`) — the direct ancestor of Plan 10.1-02's `03d4679` fix in `git log shell.html`. No manual file editing; no template mutation by hand; no orphaned stash. The cycle's git-history audit trail is identical to what a stash-based cycle would leave behind.
- **Precedent:** Plan 10.1-02's SUMMARY §Forward References anticipated exactly this mechanism: "That step can be reproduced via `git checkout c923991 -- forge_bridge/console/templates/shell.html`, run test (expect FAIL), restore with `git checkout HEAD -- forge_bridge/console/templates/shell.html`, run test (expect PASS)." Plan 10.1-04's own action block appears to have been drafted before that forward-reference update was internalized; the two planning artifacts are now reconciled in favor of the mechanism that actually works.
- **Files modified:** None beyond Task 1 (pyproject.toml) + Task 2 (new test file). The cycle itself left no residual working-tree changes.
- **Commit:** None needed (deviation is procedural; no code change resulted).
