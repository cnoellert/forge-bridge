concrete table projekt-forge's _persist_execution adapter writes into.

NOTE: Hand-written migration — Alembic autogenerate does not reliably detect
composite UNIQUE constraints or index direction (DESC).

Revision ID: 005
Revises: 004
"""

from alembic import op

revision: str = "005"
down_revision: str = "004"
branch_labels: Union[str, Sequence[str], None] = None

        sa.Column("timestamp", sa.DateTime(timezone=True), nullable=False),
        sa.Column("raw_code", sa.Text(), nullable=False),
        sa.Column("intent", sa.Text(), nullable=True),
        sa.UniqueConstraint(
            "code_hash",
            "timestamp",
            name="uq_execution_log_code_hash_timestamp",
        ),
    )

        "ix_execution_log_code_hash",
        "execution_log",

    )

        "ix_execution_log_timestamp",

def _persist_execution(record: ExecutionRecord) -> None:
    """Storage callback: mirror ExecutionRecord to the execution_log SQL table (STORE-05).

    Contract:
      * Sync (D-07): safe to call from Flame threads — no event loop required.
      * No retry (D-06 / P-03.5): DB outage is logged at WARNING once; JSONL
        is the source of truth (D-05). Retry would stack connection attempts
        and trigger QueuePool exhaustion under sustained outage.
      * Idempotent (D-09): ON CONFLICT DO NOTHING on (code_hash, timestamp).
        Two forge-bridge processes sharing this DB produce no duplicate rows.
      * Bound parameters (security): SQLAlchemy Core insert(...).values(...)
        binds raw_code via psycopg2 parameter substitution — never string
        concatenation. Neutralizes SQL injection via user-submitted Python
        in raw_code.
      * No credential leaks (security): on exception, logs only the first
        12 chars of code_hash and the exception class — NEVER the engine URL.

    Satisfies forge_bridge.StoragePersistence (asserted via isinstance at
    registration in init_learning_pipeline — D-11).
    """
    try:
        engine = _get_sync_engine()
        stmt = (
            pg_insert(_execution_log_table)
            .values(
                code_hash=record.code_hash,
                timestamp=record.timestamp,
                raw_code=record.raw_code,
                intent=record.intent,
            )
            .on_conflict_do_nothing(index_elements=["code_hash", "timestamp"])
        )
        with engine.begin() as conn:
            conn.execute(stmt)
    except Exception as exc:
        # D-06 no-retry invariant: log ONCE at WARNING, swallow.
        # SECURITY: do NOT log the engine URL (contains credentials).
        # Log only the code_hash prefix + exception class + message.
        logger.warning(
            "execution_log DB write failed — JSONL unaffected "
            "(code_hash=%s, error=%s: %s)",
            record.code_hash[:12],
            type(exc).__name__,
            exc,
        )
